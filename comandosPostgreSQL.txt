CREATE FUNCTION func_listar_person_TABLA()
RETURNS SETOF person AS
$BODY$
DECLARE
    reg RECORD;
BEGIN
    FOR REG IN SELECT * FROM person LOOP
       RETURN NEXT reg;
    END LOOP;
    RETURN;
END
$BODY$ LANGUAGE 'plpgsql'
